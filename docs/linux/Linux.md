# Linux

```sh
# 创建文件夹，此处创建一个叫linux的文件夹
mkdir linux

# 创建文件，此处创建一个index.js文件
touch index.js

# 进入目录
cd linux

# 创建并进入文件夹
mkdir linux && cd linux

# 返回上级目录
cd ..

# 返回根目录
cd /

# 查看当前路径
pwd

# 查看当前目录下的子目录和文件
ls

# 打开文件


# 清空窗口内容
clear

# 查看历史记录
history
```
